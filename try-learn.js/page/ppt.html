关于PPT方式显示数据

1、在页面添加需要的元素
2、在元素的class加上【fragment】元素就默认隐藏掉了
3、点击下一步，每点击一下显示一个元素

代码演示


function linkify( selector ) {
  if( supports3DTransforms ) {

    var nodes = document.querySelectorAll( selector );

    for( var i = 0, len = nodes.length; i < len; i++ ) {
      var node = nodes[i];

      if( !node.className ) {
        node.className += ' roll';
      }
    }
  }
}
